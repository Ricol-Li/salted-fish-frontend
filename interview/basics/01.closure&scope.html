# section 10

TIP

考察：匿名函数具名化「复合官方语法规范，推荐使用」，是指原本匿名函数没有名字，我们给其设置一个名字

    1. 我们给匿名函数设置的名字，并不会再其所处的上下文中声明，所以在函数外面不可以使用这个名字进行相关操作
    1. 但是会在自己执行产生的私有上下文中声明这个函数名，存储的值是当前这个匿名函数本身
    1. 并且默认是不支持把其值进行修改的「改了也没用」
    1. 但是如果当前私有上下文中，也有基于其余的方式声明过这个名字，则以自己声明的为主「也就是第二步做的事情优先级最低」
// 1. 不修改
;(function func(x) {
  console.log(func) // f func(x) {}
  func = 100
  console.log(func)
})(10)
// console.log(func) func is not defined
function func2(x) {}
console.log(funcs) // f func2(x){}

// 2.可修改
;(function func(x) {
  console.log(func) // undefined
  var func = 100
  console.log(func) // 100
})(10)

匿名函数递归

  • 可以实现匿名函数的递归处理
  • 语法更规范，也无需担心会和外面的变量产生冲突
// 非严格模式下
let total = (function (num) {
  if (num === 1) return num
  return (num += arguments.callee(num - 1))
})(100)

// 严格模式下
// Uncaught TypeError: 'caller', 'callee', and 'arguments' properties
// may not be accessed on strict mode functions
// or the arguments objects for calls to them
;('use strict')
let total = (function anonymous(num) {
  if (num === 1) return num
  return num + anonymous(num - 1)
})(100)
console.log(total)

{
  function foo() {}

// 添加测试代码

  console.log(foo) // f functon foo(){}
  function foo() {} // 赋值，把全局的foo改成函数，断开和全局的联系

  console.log(foo) // 1
}

# 示例 2

// 块级作用域下的函数，在全局只提升，不赋值
console.log(foo)
{
  // 块级作用域的函数，在当前作用域下，既提升又赋值
  console.log(foo) // f foo() {//b} 输出函数
  function foo() {
    // 此函数声明只针对全局，不针对私有
    // 只赋值操作
    // a
  }
  foo = 1 // 私有foo改为1
  function foo() {
    // 此函数声明只针对全局，不针对私有
    // 只对全局进行赋值操作，但是此时foo = 1了 所以全局的foo = 1
    // b
  }
}
console.log(foo)

TIP

  1. 只要函数有 ES6 形参赋值默认值「无论是否生效」
  2. 函数体中出现基于 let/const/var 声明变量的操作「没有 function，无论声明的是什么变量」

如果这两个条件同时成立，则函数执行会产生两个上下文：

  1. 函数私有上下文「形参赋值完即结束」
  2. 会把函数体当做一个块级私有上下文，并且上级上下文就是函数这个私有上下文，并且，如果块级中的某个变量和函数私有上下文中的形参是一个名字，则默认会把函数私有上下文中的形参的值，默认同步给块中的这个上下文一份
debugger
var x = 1
function func(
  x,
  y = function () {
    x = 2
  }
) {
  var x = 3
  var y = function () {
    x = 4
  }
  y()
  console.log(x)
}
func(5)
console.log(x)
