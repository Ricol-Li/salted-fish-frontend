# JS 基础面试题-this

  • 事件绑定
  • 函数执行（包括自执行函数 & 回调函数）
  • new 构造函数
  • 箭头函数

# section 1

// debugger
var num = 10
var obj = {
  num: 20,
}
obj.fn = (function (num) {
  // 自执行函数的this 是window/undefined
  this.num = num * 3 // window.num = 20 * 3 = 60
  num++ // 21
  return function (n) {
    this.num += n
    num++
    console.log(num)
  }
})(obj.num)
var fn = obj.fn
fn(5) // window.num = 60 + 5 = 65; num: 22
obj.fn(10) // obj.num = 20 + 10 = 30; num: 23
console.log(num, obj.num) // 65 30

# section 3

window.val = 1
var json = {
  val: 10,
  dbl: function () {
    this.val *= 2 //this.val = this.val * 2;
  },
}
json.dbl()
// this->json
// this.val = this.val * 2;  -> json.val = 20;
var dbl = json.dbl
dbl()
// this->window
// this.val = this.val * 2;  -> window.val = 2;
json.dbl.call(window)
// this->window  基于call强制改变函数中的this
// this.val = this.val * 2;  -> window.val = 4;
console.log(window.val + json.val) //"24"
