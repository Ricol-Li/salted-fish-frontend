# call&apply&bind

TIP

Function.prototype.call = function(){}

Function.prototype.apply = function(){}

Function.prototype.bind = function(){}

所有的函数都是 Function 类的实例，所以所有函数都可以调用者三个方法；而这三个方法都是原来改变函数中的 this 指向的

  • call vs apply

    • 都是把函数立即执行，改变函数中的 this 指向
    • 唯一区别：apply 要求把传递给函数的实参，以数组的形式管理起来，最终效果和 call 一样，也是把数组中的每一项，一个个的传给函数
    • 真实项目中建议大家使用 call，因为其性能好一些（做过测试：三个及以上参数，call 的性能明显比 apply 好一些）

# call 和 apply 的使用

const fn = function fn(x, y) {
  console.log(this, x, y)
}

let obj = {
  name: '张三',
}

// obj.fn(10,20) // Uncaught TypeError: obj.fn is not a function
// 这时我们可以使用call 和 bind 改变this指向
fn(10, 20) // this->window 10 20
fn.call(obj, 10, 20) // this->obj 10 20
fn.apply(obj, [10, 20]) // this->obj 10 20

// @1 fn基于__proto__找到Function.prototype.call，把call方法执行
// @2 在call方法执行的时候：
//    +

// 其他小细节
fn.call(10, 20) // this->new Number(10) x->20 y->undefined
fn.call() // this->window/undefined(严格模式)
fn.call(null) // this->window/null(严格模式)

# 解析 call 原理

Function.prototype.call = function (context, ...params) {
  // 判断context是null
  if (context === null) context = window
  // 判断context是不是原始值，变原始值
  if (!/^(object|function).test(typeof context)/) {
    context = Object(context)
  }
  // this -> fn
  // context -> obj
  // 思路就是让context设置一个属性xxx，让属性值等于要执行的函数（即this「即fn」）
  // 后面就可以基于context.xxx()执行
  // 这样既把函数执行了，也让函数的this改成context了
  // 为了新增一个属性不和原有属性冲突，可以设置Symbol
  let self = this
  let key = Symbol('key')
  let result

  context[key] = self
  result = context[key](...params)

  // delete context[key] // 新增的属性用完记得删除
  Reflect.deleteProperty(context, 'key')
  return result
}
const fn = function fn(x, y) {
  console.log(this, x, y)
  return x + y
}

let obj = {
  name: '张三',
}
console.log(fn.call(obj, 10, 20))
// @1 fn基于__proto__找到Function.prototype.call，把call方法执行
// @2 在call方法执行的时候：
// context:obj 要改变的this指向，params:[10,20] 执行函数传递的实参信息
// this:fn 要执行的函数
// 它干的事情是：立即把fn执行，并且让fn中的this指向obj，把10/20传递给fn
// 接受fn执行的返回结果，作为最后的结果返回

# call vs bind

call 是把函数立即执行，而 bind 只是预处理函数中的 this 和参数，函数此时并没有执行

# 实现 bind

Function.prototype.myBind = function myBind(context, ...params) {
  // this -> fn context -> obj
  const _this = this
  return function proxy(...args) {
    params = params.concat(args)
    return _this.call(context, ...params)
  }
}

const fn = function fn(x, y) {
  console.log(this, x, y)
  return x + y
}

let obj = {
  name: '张三',
}
console.log(fn.myBind(obj, 10, 20)())
